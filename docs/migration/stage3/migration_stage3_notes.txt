1. ICON_TYPES MODULE (icon_types.c / icon_types.h)
================================================================================

PURPOSE:
--------
Handles icon type detection, size queries, and format identification for
standard icons, NewIcons, and OS3.5 ColorIcons.

KEY CHANGES:
------------
1. Modern icon.library API usage:
   - GetDiskObject(CONST_STRPTR name) → struct DiskObject*
   - FreeDiskObject(struct DiskObject*) → void
   - Proper error checking with NULL return values

2. BPTR consistency:
   - All file handles changed from void* to BPTR
   - Lock() returns BPTR, initialized to 0 (not NULL)
   - Proper Lock/UnLock pairing verified

3. BOOL normalization:
   - Public API functions return BOOL (TRUE/FALSE)
   - Internal logic may use bool for convenience
   - Consistent with AmigaDOS conventions

4. Error handling:
   - IoErr() checked after failed GetDiskObject()
   - append_to_log() used for debugging and error reporting
   - Graceful fallback to default sizes

CRITICAL FUNCTIONS:
-------------------

GetNewIconSizePath(const char *filePath, IconSize *newIconSize)
----------------------------------------------------------------
Purpose: Extract NewIcon size from IM1= tooltype
Changes:
  - Proper .info suffix removal with bounds checking
  - whd_malloc() for temporary path copy
  - GetDiskObject() with NULL check
  - ToolTypes array iteration with NULL terminator check
  - FreeDiskObject() and whd_free() on all paths

GetStandardIconSize(const char *filePath, IconSize *iconSize)
--------------------------------------------------------------
Purpose: Get standard icon size from DiskObject gadget dimensions
Changes:
  - BPTR return type for consistency
  - GetDiskObject() with error checking
  - Icon size extracted from do_Gadget.Width/Height
  - FreeDiskObject() cleanup
  - Returns TRUE/FALSE (BOOL)

IsNewIconPath(const STRPTR filePath)
------------------------------------
Purpose: Detect if icon file contains NewIcons data
Changes:
  - BPTR for file handle (changed from void*)
  - Open(MODE_OLDFILE) / Close() pairing
  - ToolTypes iteration for IM1= detection
  - FreeDiskObject() on all exit paths
  - Returns BOOL

isOS35IconFormat(const char *filename)
--------------------------------------
Purpose: Detect OS3.5+ ColorIcon format (PNG-based)
Changes:
  - BPTR for file handle
  - Direct file reading with Read()
  - Magic number check: 0xE310000 at offset 0x14
  - Proper Close() on all paths
  - Returns TRUE/FALSE (BOOL)

GetIconSizeFromFile(const char *filePath, IconSize *iconSize)
--------------------------------------------------------------
Purpose: Universal icon size getter with format detection
Changes:
  - Tries NewIcon size first (IM1= tooltype)
  - Falls back to standard icon size
  - Then tries OS3.5 format
  - Logs failures via append_to_log()
  - Returns BOOL

VBCC-SPECIFIC NOTES:
--------------------
- VBCC accepts inline keyword for small helpers
- Mixed declarations allowed (C99)
- // comments fully supported
- snprintf() available in VBCC's C99 mode
- struct DiskObject defined in <workbench/icon.h>
- Proto includes: <proto/icon.h>, <proto/dos.h>

CODE EXAMPLE:
-------------
/* VBCC MIGRATION NOTE (Stage 3): Modern icon.library usage */
struct DiskObject *diskObject = GetDiskObject(filePath);
if (!diskObject) {
    append_to_log("Failed to load icon: %s (IoErr: %ld)\n", filePath, IoErr());
    return FALSE;
}

/* Extract size from gadget */
iconSize->width = diskObject->do_Gadget.Width;
iconSize->height = diskObject->do_Gadget.Height;

/* Always free resources */
FreeDiskObject(diskObject);
return TRUE;

DEBUGGING:
----------

    append_to_log("Loading icon: %s\n", filePath);
    append_to_log("Icon size: %d x %d\n", width, height);
#endif

    append_to_log("Using default Workbench settings (IoErr: %ld)\n", IoErr());
    InitializeDefaultWorkbenchSettings(settings);
    return;

UBYTE buffer[12];
if (Read(file, buffer, 12) == 12) {
    ULONG chunkType = *((ULONG *)(buffer + 8));
    if (chunkType == MAKE_ID('P', 'R', 'E', 'F')) {
        /* Valid preference file */
        while (Read(file, buffer, 8) == 8) {
            ULONG type = *((ULONG *)buffer);
            ULONG size = *((ULONG *)(buffer + 4));
            size = (size + 1) & ~1;  /* Even padding */

                UBYTE *data = AllocMem(size, MEMF_PUBLIC | MEMF_CLEAR);
                if (data) {
                    Read(file, data, size);
                    struct WorkbenchPrefs *prefs = (struct WorkbenchPrefs *)data;
                    settings->borderless = prefs->wbp_Borderless;
                    FreeMem(data, size);
                }
            }
        }
    }
}

Close(file);

CRITICAL FUNCTIONS:
-------------------

-----------------
Purpose: Return default font preference
Changes:
  - malloc(sizeof(FontPref))
  - strncpy("topaz.font", ...)
  - size = 8
  - Returns FontPref*

getIconFont(void)
-----------------
Purpose: Get Workbench icon font preference
Changes:
  - defaultFont() as initial value
  - Checks ENV:sys/font.prefs first (WB 3.x)
  - Falls back to ENV:sys/wbfont.prefs (WB 2.x)
  - extractFonts() for parsing

extractFonts(const char *filename)
----------------------------------
Purpose: Parse font preference file
Changes:
  - BPTR file = Open(filename, MODE_OLDFILE)
  - UBYTE buffer[FONT_FILE_BUFFER] for reading
  - Read() in chunks
  - Search for "FONT\0\0" marker
  - Extract fontSize (endianness-aware)
  - Extract fontName (36 bytes after marker)
  - malloc() for FontPref result
  - Close() file

VBCC-SPECIFIC NOTES:
--------------------
- FONT chunk format varies between WB 2.x and 3.x
- font.prefs uses little-endian 16-bit size
- wbfont.prefs uses big-endian 8.8 fixed-point size
- Fixed-point conversion: fontSize = fixedSize / 256
- Font name is null-terminated string

CODE EXAMPLE:
-------------
/* VBCC MIGRATION NOTE (Stage 3): Font preference extraction */
FontPref *extractFonts(const char *filename) {
    BPTR file = Open(filename, MODE_OLDFILE);
    if (!file) {
        return defaultFont();
    }
    
    UBYTE buffer[256];
    LONG bytesRead;
    FontPref *result = NULL;
    
    while ((bytesRead = Read(file, buffer, 256)) > 0 && !result) {
        for (int i = 0; i < bytesRead - 40; i++) {
            if (buffer[i] == 'F' && buffer[i+1] == 'O' &&
                buffer[i+2] == 'N' && buffer[i+3] == 'T' &&
                buffer[i+4] == 0x00 && buffer[i+5] == 0x00) {
                
                /* Found FONT marker */
                unsigned int fontSize;
                if (useOldMethod) {
                    /* WB 3.x: little-endian 16-bit */
                    fontSize = buffer[i+33] | (buffer[i+34] << 8);
                } else {
                    /* WB 2.x: big-endian 8.8 fixed-point */
                    unsigned short fixedSize = (buffer[i+33] << 8) | buffer[i+34];
                    fontSize = fixedSize / 256;
                }
                
                /* Extract font name */
                char fontName[32];
                int j;
                for (j = 0; j < 31 && buffer[i+36+j] != 0; j++) {
                    fontName[j] = buffer[i+36+j];
                }
                fontName[j] = '\0';
                
                /* Create result */
                result = malloc(sizeof(FontPref));
                if (result) {
                    strncpy(result->name, fontName, sizeof(result->name));
                    result->size = fontSize;
                }
                break;
            }
        }
    }
    
    Close(file);
    return result ? result : defaultFont();
}

DEBUGGING:
